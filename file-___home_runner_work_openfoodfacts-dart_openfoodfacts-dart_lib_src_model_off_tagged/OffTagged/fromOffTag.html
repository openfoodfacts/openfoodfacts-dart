Implementation

static OffTagged? fromOffTag(
  final String? offTag,
  final Iterable<OffTagged> values,
) {
  if (offTag == null) {
    return null;
  }
  for (final OffTagged offTagged in values) {
    if (offTagged.offTag == offTag) {
      return offTagged;
    }
  }
  return null;
}
