MediaType? imagineMediaType(final String filename) {
  String ext = extension(filename);
  if (ext.isEmpty) {
    return null;
  }
  ext = ext.substring(1).toLowerCase();
  switch (ext) {
    case 'jpg':
    case 'jpeg':
      return MediaType('image', 'jpeg');
    case 'png':
      return MediaType('image', 'png');
    case 'webp':
      return MediaType('image', 'webp');
  }
  return null;
}
