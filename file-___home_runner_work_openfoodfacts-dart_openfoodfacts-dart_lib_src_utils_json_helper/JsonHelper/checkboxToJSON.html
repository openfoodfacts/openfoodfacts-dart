static String? checkboxToJSON(dynamic value) {
  if (value == null) {
    return null;
  } else if (value == true ||
      (value is String && value.trim().toLowerCase() == _checkboxOnValue)) {
    return _checkboxOnValue;
  } else {
    return _checkboxOffValue;
  }
}
