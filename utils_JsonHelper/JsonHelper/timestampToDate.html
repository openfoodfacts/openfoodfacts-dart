static DateTime? timestampToDate(dynamic json) {
  if (json == null) {
    return null;
  }
  final int timestamp = JsonObject.parseInt(json)!;
  return DateTime.fromMillisecondsSinceEpoch(
      Duration.millisecondsPerSecond * timestamp,
      isUtc: true);
}
